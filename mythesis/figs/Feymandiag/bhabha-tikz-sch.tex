\documentclass{standalone}

\usepackage{tikz}
\usepackage{pgfplots}
\usetikzlibrary{positioning,shapes,arrows}
\usetikzlibrary{decorations.pathmorphing}
\usetikzlibrary{decorations.markings}

\begin{document}
\tikzset{
  particle/.style={thick, draw=black, postaction={decorate},
    decoration={markings, mark=at position 0.6 with {\arrow[black]{triangle 45}}}},
    photon/.style={thick,decorate, draw=black,
    decoration={coil, aspect=0}}
}

\begin{tikzpicture}[node distance=1cm and 1.5cm]
  \coordinate[label=left:$e^{-}$] (e1);
  \coordinate[below=3cm of e1, label=left:$e^{+}$] (e2);
  \coordinate[below right=1.5cm and 1.5cm of e1] (aux); % Interaction point

  \draw[particle] (e1) -- (aux) node[midway, above left] {};
  \draw[particle] (aux) -- (e2) node[midway, below left] {};
  \draw[photon]   (aux) -- +(0:2cm) node[midway, above] {$\gamma$};

  \coordinate[right=2cm of aux] (aux2); % Decay point of photon/Z0
  \coordinate[above right=1.5cm and 1cm of aux2, label=right:$e^{-}$] (e3);
  \coordinate[below right=1.5cm and 1cm of aux2, label=right:$e^{+}$] (e4);

  \draw[particle] (aux2) -- (e3);
  \draw[particle] (e4) -- (aux2);
\end{tikzpicture}

\end{document}
